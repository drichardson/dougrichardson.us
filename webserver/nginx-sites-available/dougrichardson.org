#
# Redirect to github page to prepare for no longer owning delicioussafari.com.
#
server {
    listen 80;
    listen [::]:80;
    server_name dougrichardson.org;
    return 301 https://dougrichardson.us$request_uri;
}

server {
    # for debugging only
    # error_log	/var/log/nginx/dougrichardson.org.error.log debug;

    #
    # SSL configuration
    #
    listen 443 ssl;
    listen [::]:443 ssl;

    add_header Content-Security-Policy "default-src 'self'; block-all-mixed-content;";

    server_name dougrichardson.org;

    expires 5m;

    return 301 https://dougrichardson.us$request_uri;

    ssl_certificate /etc/letsencrypt/live/dougrichardson.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/dougrichardson.org/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/dougrichardson.org/chain.pem;
}
